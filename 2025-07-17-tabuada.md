    &lt;script&gt;
        // Variáveis do jogo
        let currentMode = '';
        let score = 0;
        let startTime;
        let timerInterval;
        let currentQuestion = {};
        let questionsAnswered = 0;
        const totalQuestions = 10;
        
        // Inicia o jogo
        function startGame(mode) {
            currentMode = mode;
            score = 0;
            questionsAnswered = 0;
            document.getElementById('score').textContent = `Pontuação: ${score}`;
            document.getElementById('progress').style.width = '0%';
            
            if (mode === 'desafio') {
                document.getElementById('game-title').textContent = 'Modo Desafio';
                startTimer();
            } else {
                document.getElementById('game-title').textContent = 'Modo Treino';
                document.getElementById('timer').textContent = 'Modo sem tempo limite';
            }
            
            document.getElementById('menu').style.display = 'none';
            document.getElementById('tables').style.display = 'none';
            document.getElementById('game').style.display = 'block';
            
            generateQuestion();
        }
        
        // Mostra as tabuadas completas
        function showTables() {
            document.getElementById('menu').style.display = 'none';
            document.getElementById('game').style.display = 'none';
            document.getElementById('tables').style.display = 'block';
            
            let tablesHTML = '';
            for (let i = 1; i &lt;= 10; i++) {
                tablesHTML += `&lt;h3&gt;Tabuada do ${i}&lt;/h3&gt;`;
                tablesHTML += '&lt;table class="tabuada-table"&gt;';
                tablesHTML += '&lt;tr&gt;&lt;th&gt;Operação&lt;/th&gt;&lt;th&gt;Resultado&lt;/th&gt;&lt;/tr&gt;';
                for (let j = 1; j &lt;= 10; j++) {
                    tablesHTML += `&lt;tr&gt;&lt;td&gt;${i} × ${j}&lt;/td&gt;&lt;td&gt;${i*j}&lt;/td&gt;&lt;/tr&gt;`;
                }
                tablesHTML += '&lt;/table&gt;';
            }
            
            document.getElementById('tables-container').innerHTML = tablesHTML;
        }
        
        // Volta ao menu principal
        function backToMenu() {
            document.getElementById('menu').style.display = 'block';
            document.getElementById('tables').style.display = 'none';
            document.getElementById('game').style.display = 'none';
        }
        
        // Finaliza o jogo
        function endGame() {
            if (currentMode === 'desafio') {
                clearInterval(timerInterval);
            }
            backToMenu();
        }
        
        // Inicia o cronômetro para o modo desafio
        function startTimer() {
            startTime = new Date();
            timerInterval = setInterval(updateTimer, 1000);
        }
        
        // Atualiza o cronômetro
        function updateTimer() {
            const currentTime = new Date();
            const elapsed = Math.floor((currentTime - startTime) / 1000);
            document.getElementById('timer').textContent = `Tempo: ${elapsed}s`;
        }
        
        // Gera uma nova pergunta
        function generateQuestion() {
            const num1 = Math.floor(Math.random() * 10) + 1;
            const num2 = Math.floor(Math.random() * 10) + 1;
            currentQuestion = {
                num1: num1,
                num2: num2,
                answer: num1 * num2
            };
            
            document.getElementById('question').textContent = `Quanto é ${num1} × ${num2}?`;
            document.getElementById('answer').value = '';
            document.getElementById('feedback').textContent = '';
            document.getElementById('answer').focus();
        }
        
        // Verifica a resposta do usuário
        function checkAnswer() {
            const userAnswer = parseInt(document.getElementById('answer').value);
            const feedbackElement = document.getElementById('feedback');
            
            if (isNaN(userAnswer)) {
                feedbackElement.textContent = 'Por favor, digite um número válido!';
                feedbackElement.className = 'feedback incorrect';
                return;
            }
            
            questionsAnswered++;
            document.getElementById('progress').style.width = `${(questionsAnswered / totalQuestions) * 100}%`;
            
            if (userAnswer === currentQuestion.answer) {
                score += (currentMode === 'desafio' ? 2 : 1);
                document.getElementById('score').textContent = `Pontuação: ${score}`;
                feedbackElement.textContent = 'Correto! Parabéns!';
                feedbackElement.className = 'feedback correct';
            } else {
                feedbackElement.textContent = `Ops! A resposta correta era ${currentQuestion.answer}.`;
                feedbackElement.className = 'feedback incorrect';
            }
            
            if (questionsAnswered &lt; totalQuestions) {
                setTimeout(generateQuestion, 1500);
            } else {
                setTimeout(() =&gt; {
                    feedbackElement.textContent += ` Jogo concluído! Sua pontuação final: ${score}`;
                    if (currentMode === 'desafio') {
                        const elapsed = parseInt(document.getElementById('timer').textContent.replace('Tempo: ', '').replace('s', ''));
                        feedbackElement.textContent += ` em ${elapsed} segundos.`;
                    }
                }, 500);
            }
        }
        
        // Permite responder pressionando Enter
        document.getElementById('answer').addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                checkAnswer();
            }
        });
    &lt;/script&gt;
