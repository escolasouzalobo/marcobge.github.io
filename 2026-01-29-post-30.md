    &lt;script&gt;
        // Elementos do DOM
        const proportionalityTypeSelect = document.getElementById('proportionalityType');
        const constantValueSlider = document.getElementById('constantValue');
        const constantValueDisplay = document.getElementById('constantValueDisplay');
        const dataPointsSlider = document.getElementById('dataPoints');
        const dataPointsDisplay = document.getElementById('dataPointsDisplay');
        const resetButton = document.getElementById('resetButton');
        const randomButton = document.getElementById('randomButton');
        const tableBody = document.getElementById('tableBody');
        const explanationType = document.getElementById('explanationType');
        const explanationText = document.getElementById('explanationText');
        const formulaDisplay = document.getElementById('formulaDisplay');
        const graphSVG = document.getElementById('proportionalityGraph');
        
        // Configuração inicial
        let proportionalityType = 'direct';
        let constantValue = 5;
        let dataPoints = 6;
        
        // Inicializar applet
        function initApplet() {
            updateDisplays();
            updateFormula();
            updateExplanation();
            generateTable();
            drawGraph();
            
            // Configurar event listeners
            proportionalityTypeSelect.addEventListener('change', function() {
                proportionalityType = this.value;
                updateFormula();
                updateExplanation();
                generateTable();
                drawGraph();
            });
            
            constantValueSlider.addEventListener('input', function() {
                constantValue = parseInt(this.value);
                updateDisplays();
                updateFormula();
                generateTable();
                drawGraph();
            });
            
            dataPointsSlider.addEventListener('input', function() {
                dataPoints = parseInt(this.value);
                updateDisplays();
                generateTable();
                drawGraph();
            });
            
            resetButton.addEventListener('click', function() {
                proportionalityTypeSelect.value = 'direct';
                proportionalityType = 'direct';
                constantValueSlider.value = 5;
                constantValue = 5;
                dataPointsSlider.value = 6;
                dataPoints = 6;
                updateDisplays();
                updateFormula();
                updateExplanation();
                generateTable();
                drawGraph();
            });
            
            randomButton.addEventListener('click', function() {
                constantValue = Math.floor(Math.random() * 15) + 2;
                constantValueSlider.value = constantValue;
                proportionalityType = Math.random() &gt; 0.5 ? 'direct' : 'inverse';
                proportionalityTypeSelect.value = proportionalityType;
                updateDisplays();
                updateFormula();
                updateExplanation();
                generateTable();
                drawGraph();
            });
        }
        
        // Atualizar exibições de valores
        function updateDisplays() {
            constantValueDisplay.textContent = constantValue;
            dataPointsDisplay.textContent = dataPoints;
        }
        
        // Atualizar fórmula exibida - USANDO NOTAÇÃO SIMPLES
        function updateFormula() {
            if (proportionalityType === 'direct') {
                if (constantValue === 1) {
                    formulaDisplay.textContent = 'y = x';
                } else {
                    // Usando asterisco como operador de multiplicação
                    formulaDisplay.textContent = `y = ${constantValue} * x`;
                }
            } else {
                // Usando barra como operador de divisão
                formulaDisplay.textContent = `y = ${constantValue} / x`;
            }
        }
        
        // Atualizar explicação - CRIANDO DINAMICAMENTE
        function updateExplanation() {
            if (proportionalityType === 'direct') {
                explanationType.textContent = 'Proporcionalidade Direta';
                explanationType.className = 'proportionality-type direct';
                
                // Criar explicação diretamente no DOM para evitar problemas de codificação
                let formulaText = constantValue === 1 ? 'y = x' : `y = ${constantValue} * x`;
                
                explanationText.innerHTML = `
                    &lt;p&gt;Na &lt;strong&gt;proporcionalidade direta&lt;/strong&gt;, duas grandezas estão relacionadas de forma que quando uma aumenta, a outra também aumenta na mesma proporção. A relação é expressa por &lt;strong&gt;${formulaText}&lt;/strong&gt;, onde &lt;strong&gt;k = ${constantValue}&lt;/strong&gt; é a constante de proporcionalidade. No gráfico, essa relação é representada por uma linha reta que passa pela origem (0,0).&lt;/p&gt;
                    &lt;p&gt;&lt;strong&gt;Características:&lt;/strong&gt;&lt;/p&gt;
                    &lt;ul&gt;
                        &lt;li&gt;A razão y/x é sempre constante (igual a ${constantValue})&lt;/li&gt;
                        &lt;li&gt;O gráfico é uma linha reta&lt;/li&gt;
                        &lt;li&gt;A reta passa pelo ponto (0,0)&lt;/li&gt;
                    &lt;/ul&gt;
                `;
            } else {
                explanationType.textContent = 'Proporcionalidade Inversa';
                explanationType.className = 'proportionality-type inverse';
                
                // Criar explicação diretamente no DOM para evitar problemas de codificação
                explanationText.innerHTML = `
                    &lt;p&gt;Na &lt;strong&gt;proporcionalidade inversa&lt;/strong&gt;, duas grandezas estão relacionadas de forma que quando uma aumenta, a outra diminui na mesma proporção. A relação é expressa por &lt;strong&gt;y = ${constantValue} / x&lt;/strong&gt; ou &lt;strong&gt;x * y = ${constantValue}&lt;/strong&gt;, onde &lt;strong&gt;k = ${constantValue}&lt;/strong&gt; é a constante de proporcionalidade. No gráfico, essa relação é representada por uma curva chamada hipérbole.&lt;/p&gt;
                    &lt;p&gt;&lt;strong&gt;Características:&lt;/strong&gt;&lt;/p&gt;
                    &lt;ul&gt;
                        &lt;li&gt;O produto x*y é sempre constante (igual a ${constantValue})&lt;/li&gt;
                        &lt;li&gt;O gráfico é uma curva (hipérbole)&lt;/li&gt;
                        &lt;li&gt;A curva nunca toca os eixos&lt;/li&gt;
                    &lt;/ul&gt;
                `;
            }
        }
        
        // Gerar tabela de valores
        function generateTable() {
            tableBody.innerHTML = '';
            
            for (let x = 1; x &lt;= dataPoints; x++) {
                const row = document.createElement('tr');
                
                // Calcular y baseado no tipo de proporcionalidade
                let y;
                if (proportionalityType === 'direct') {
                    y = constantValue * x;
                } else {
                    y = constantValue / x;
                }
                
                // Calcular razão e produto
                const ratio = (y / x).toFixed(2);
                const product = (x * y).toFixed(2);
                
                // Adicionar células à linha - usando texto simples
                row.innerHTML = `
                    &lt;td&gt;${x}&lt;/td&gt;
                    &lt;td&gt;${y.toFixed(2)}&lt;/td&gt;
                    &lt;td&gt;${ratio}&lt;/td&gt;
                    &lt;td&gt;${product}&lt;/td&gt;
                `;
                
                tableBody.appendChild(row);
            }
        }
        
        // Desenhar gráfico SVG
        function drawGraph() {
            // Limpar o SVG
            graphSVG.innerHTML = '';
            
            // Dimensões do gráfico
            const width = 400;
            const height = 220;
            const padding = 30;
            const graphWidth = width - 2 * padding;
            const graphHeight = height - 2 * padding;
            
            // Calcular valores
            const values = [];
            let maxX = dataPoints;
            let maxY = 0;
            
            for (let x = 1; x &lt;= dataPoints; x++) {
                let y;
                if (proportionalityType === 'direct') {
                    y = constantValue * x;
                } else {
                    y = constantValue / x;
                }
                values.push({x, y});
                
                if (y &gt; maxY) maxY = y;
            }
            
            // Ajustar máximos para melhor visualização
            if (proportionalityType === 'inverse') {
                maxY = constantValue * 1.2;
            } else {
                maxY = Math.ceil(maxY * 1.1);
            }
            
            // Função para mapear valores para coordenadas SVG
            const mapX = (x) =&gt; padding + (x / maxX) * graphWidth;
            const mapY = (y) =&gt; height - padding - (y / maxY) * graphHeight;
            
            // Desenhar eixos
            // Eixo X
            graphSVG.innerHTML += '&lt;line x1="' + padding + '" y1="' + (height - padding) + '" x2="' + (width - padding) + '" y2="' + (height - padding) + '" stroke="#333" stroke-width="2" /&gt;';
            // Eixo Y
            graphSVG.innerHTML += '&lt;line x1="' + padding + '" y1="' + padding + '" x2="' + padding + '" y2="' + (height - padding) + '" stroke="#333" stroke-width="2" /&gt;';
            
            // Marcas no eixo X
            for (let i = 0; i &lt;= maxX; i += Math.max(1, Math.floor(maxX/5))) {
                const x = mapX(i);
                graphSVG.innerHTML += '&lt;line x1="' + x + '" y1="' + (height - padding - 5) + '" x2="' + x + '" y2="' + (height - padding + 5) + '" stroke="#333" stroke-width="1" /&gt;';
                graphSVG.innerHTML += '&lt;text x="' + x + '" y="' + (height - padding + 15) + '" text-anchor="middle" font-size="10"&gt;' + i + '&lt;/text&gt;';
            }
            
            // Marcas no eixo Y
            for (let i = 0; i &lt;= maxY; i += Math.max(1, Math.floor(maxY/5))) {
                const y = mapY(i);
                graphSVG.innerHTML += '&lt;line x1="' + (padding - 5) + '" y1="' + y + '" x2="' + (padding + 5) + '" y2="' + y + '" stroke="#333" stroke-width="1" /&gt;';
                graphSVG.innerHTML += '&lt;text x="' + (padding - 10) + '" y="' + (y + 3) + '" text-anchor="end" font-size="10"&gt;' + i + '&lt;/text&gt;';
            }
            
            // Rótulos dos eixos
            graphSVG.innerHTML += '&lt;text x="' + (width/2) + '" y="' + (height - 5) + '" text-anchor="middle" font-size="12" font-weight="bold"&gt;x&lt;/text&gt;';
            graphSVG.innerHTML += '&lt;text x="10" y="' + (height/2) + '" text-anchor="middle" font-size="12" font-weight="bold" transform="rotate(-90, 10, ' + (height/2) + ')"&gt;y&lt;/text&gt;';
            
            // Desenhar linha/pontos
            if (proportionalityType === 'direct') {
                // Linha para proporcionalidade direta
                let pathData = 'M ' + mapX(0) + ' ' + mapY(0) + ' ';
                for (let i = 0; i &lt;= maxX; i += 0.1) {
                    const y = constantValue * i;
                    if (y &lt;= maxY) {
                        pathData += 'L ' + mapX(i) + ' ' + mapY(y) + ' ';
                    }
                }
                graphSVG.innerHTML += '&lt;path d="' + pathData + '" fill="none" stroke="#3498db" stroke-width="3" /&gt;';
            } else {
                // Para proporcionalidade inversa, desenhar uma curva suave
                let pathData = 'M ' + mapX(0.5) + ' ' + mapY(constantValue / 0.5) + ' ';
                for (let i = 0.5; i &lt;= maxX; i += 0.1) {
                    const y = constantValue / i;
                    if (y &lt;= maxY &amp;&amp; y &gt; 0) {
                        pathData += 'L ' + mapX(i) + ' ' + mapY(y) + ' ';
                    }
                }
                graphSVG.innerHTML += '&lt;path d="' + pathData + '" fill="none" stroke="#e74c3c" stroke-width="3" /&gt;';
            }
            
            // Desenhar pontos de dados
            const pointColor = proportionalityType === 'direct' ? '#2980b9' : '#c0392b';
            values.forEach(point =&gt; {
                const cx = mapX(point.x);
                const cy = mapY(point.y);
                graphSVG.innerHTML += '&lt;circle cx="' + cx + '" cy="' + cy + '" r="5" fill="' + pointColor + '" stroke="white" stroke-width="1" /&gt;';
            });
            
            // Título do gráfico
            const graphType = proportionalityType === 'direct' ? 'Direta' : 'Inversa';
            graphSVG.innerHTML += '&lt;text x="' + (width/2) + '" y="20" text-anchor="middle" font-size="12" font-weight="bold"&gt;Proporcionalidade ' + graphType + '&lt;/text&gt;';
        }
        
        // Inicializar quando a página carregar
        window.onload = initApplet;
    &lt;/script&gt;
