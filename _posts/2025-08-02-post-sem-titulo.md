                    &lt;ul&gt;
                        &lt;li&gt;6 faces retangulares&lt;/li&gt;
                        &lt;li&gt;12 arestas&lt;/li&gt;
                        &lt;li&gt;8 vértices&lt;/li&gt;
                    &lt;/ul&gt;

                        &lt;p&gt;O paralelepípedo é um prisma cujas bases são paralelogramos. Quando todas as faces são retângulos, chamamos de paralelepípedo retângulo.&lt;/p&gt;
                        &lt;p&gt;&lt;strong&gt;Exemplo no cotidiano:&lt;/strong&gt; Caixas de sapato, livros, tijolos.&lt;/p&gt;
                    &lt;/div&gt;

                    &lt;ul&gt;
                        &lt;li&gt;1 base poligonal&lt;/li&gt;
                        &lt;li&gt;Faces laterais triangulares&lt;/li&gt;
                        &lt;li&gt;1 vértice principal (ápice)&lt;/li&gt;

                        &lt;p&gt;&lt;strong&gt;Exemplo no cotidiano:&lt;/strong&gt; Pirâmides do Egito, telhados pontiagudos.&lt;/p&gt;

                    &lt;p&gt;&lt;strong&gt;Elementos:&lt;/strong&gt;&lt;/p&gt;
                    &lt;ul&gt;
                        &lt;li&gt;2 bases paralelas e congruentes&lt;/li&gt;
                        &lt;li&gt;Faces laterais retangulares&lt;/li&gt;
                        &lt;li&gt;Arestas laterais paralelas&lt;/li&gt;

        function openTab(tabName) {
            // Esconde todos os conteúdos das abas
            var tabContents = document.getElementsByClassName("tab-content");
            for (var i = 0; i &lt; tabContents.length; i++) {
                tabContents[i].classList.remove("active");
            }
            
            // Remove a classe 'active' de todos os botões
            var tabButtons = document.getElementsByClassName("tab-button");
            for (var i = 0; i &lt; tabButtons.length; i++) {
                tabButtons[i].classList.remove("active");
            }
            
            // Mostra a aba selecionada e ativa o botão correspondente
            document.getElementById(tabName).classList.add("active");
            event.currentTarget.classList.add("active");
        }
        
        function showDetails(id) {
            var details = document.getElementById(id + "-details");
            if (details.style.display === "none") {
                details.style.display = "block";
                event.target.textContent = "Menos detalhes";
            } else {
                details.style.display = "none";
                event.target.textContent = "Mais detalhes";
            }
        }
        
        function checkQuiz() {
            // Respostas corretas
            var answers = {
                q1: "b",
                q2: "b",
                q3: "b",
                q4: "c"
            };
            
            var score = 0;
            
            // Verifica cada questão
            for (var i = 1; i &lt;= 4; i++) {
                var questionName = "q" + i;
                var selectedOption = document.querySelector('input[name="' + questionName + '"]:checked');
                var resultElement = document.getElementById("result" + i);
                
                if (selectedOption) {
                    if (selectedOption.value === answers[questionName]) {
                        resultElement.textContent = "Correto!";
                        resultElement.className = "result correct";
                        score++;
                    } else {
                        resultElement.textContent = "Incorreto. A resposta correta é " + getCorrectAnswerText(questionName, answers[questionName]);
                        resultElement.className = "result incorrect";
                    }
                    resultElement.style.display = "block";
                } else {
                    resultElement.textContent = "Por favor, selecione uma resposta.";
                    resultElement.className = "result";
                    resultElement.style.display = "block";
                }
            }
            
            // Mostra o placar final
            alert("Você acertou " + score + " de 4 perguntas!");
        }

        function resetQuiz() {
            // Limpa todas as seleções
            var radioButtons = document.querySelectorAll('input[type="radio"]');
            for (var i = 0; i &lt; radioButtons.length; i++) {
                radioButtons[i].checked = false;
            }
            
            // Esconde todos os resultados
            var results = document.getElementsByClassName("result");
            for (var i = 0; i &lt; results.length; i++) {
